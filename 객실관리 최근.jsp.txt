<script language="javascript">
    
    $(document)
    //Create(insert부분)//
    .on('click','#update',function(){
        var r1 = $('#room-name').text();
        var r2 = $('#room-div').text();
        var r3 = $('#room-expense').text();
        var r4 = $('#person').text();
        $('#room-name').val("");
        $('#room-items2').val("");
        $('#room-expense').val("");
        $('#person').val("");
        $('#selectBox').val(r1);
        $('#selectBox').val(r2);
        $('#selectBox').val(r3);
        $('#selectBox').val(r4);
        
    })
    $(document)
    //Delete(remove(delete)부분)//
    //테이블에있는 내용물을 삭제합니다.//
    .on('click','#delete',function(){
        $('table:eq(0) tr').eq(0).remove();
    })
    $(document)
    .on('click','#reset',function(){
         $('#room-name').val("");
         $('#room-div').val("");
         $('#room-expense').val("");
         $('#person').val("");
    })
    $(document)
    .on('click','#selectBox',function(){
       var s1 =  $(this).find("option:selected").text();
       $('#room-name').val(s1);
    })
</script>
